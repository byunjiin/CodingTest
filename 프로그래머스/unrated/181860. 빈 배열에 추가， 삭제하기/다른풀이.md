# 함수사용한 풀이

```
function solution(arr, flag) {
    var answer = [];
    for(let i=0; i<flag.length; i++){
        if (flag[i] === true){
            answer = answer.concat(Array(arr[i] * 2).fill(arr[i]));
        }else{
            answer = answer.slice(0,(answer.length)-arr[i])
        }
    }
    return answer;
}
```

# while문 사용한 풀이

```
function solution(arr, flag) {
    var answer = [];
    for(let i=0; i<flag.length; i++){
        let cnt = 0;
        if (flag[i] == true){
            while(cnt < arr[i]*2){
                answer.push(arr[i])
                cnt += 1;
            }
        }else{
            while(cnt < arr[i]){
                answer.pop()
                cnt += 1;
            }
        }
    }
    return answer;
}
```
